SENTENCIA SQL PARA BASE DE DATOS:

CREATE DATABASE IF NOT EXISTS usuarios;

USE usuarios;

CREATE TABLE IF NOT EXISTS usuarios (
    id INT PRIMARY KEY AUTO_INCREMENT,
    nombre VARCHAR(45),
    correo VARCHAR(45),
    contrasena VARCHAR(45)
);
